143                     var restoredDesc = { };
144                     for (var key in desc)
145                     {
146                         if (desc.hasOwnProperty (key))
147                         {
148                             var value = desc[key];
149                             var typedValue;
150                             switch (key)
151                             {
152                                 case "book":
153                                 case "name":
154                                     typedValue = [ "<string>", localize (value) ];
155                                     break;
156                                 case "bookKey":
157                                     typedValue = [ "<data>", value ];
158                                     break;
159                                 case "bookID":
160                                     typedValue = [ "<integer>", value ];
161                                     break;
162                                 case "a":
163                                 case "b":
164                                 case "black":
165                                 case "blue":
166                                 case "brightness":
167                                 case "cyan":
168                                 case "gray":
169                                 case "green":
170                                 case "luminance":
171                                 case "magenta":
172                                 case "red":
173                                 case "saturation":
174                                 case "yellowColor":
175                                     typedValue = [ "<double>", value ];
176                                     break;
177                                 case "hue":
178                                     typedValue = [ "<unitDouble>", [ "angleUnit", value ] ];
179                                     break;
180                                 case "color":
181                                     var colorClass;
182                                     if ((("book" in value) && ("name" in value)) || (("bookID" in value) && ("bookKey" in value)))
183                                     {
184                                         colorClass = "bookColor";
185                                     }
186                                     else if (("cyan" in value) && ("magenta" in value) && ("yellowColor" in value) && ("black" in value))
187                                     {
188                                         colorClass = "CMYKColorClass";
189                                     }
190                                     else if ("gray" in value)
191                                     {
192                                         colorClass = "grayscale";
193                                     }
194                                     else if (("hue" in value) && ("saturation" in value) && ("brightness" in value))
195                                     {
196                                         colorClass = "HSBColorClass";
197                                     }
198                                     else if (("luminance" in value) && ("a" in value) && ("b" in value))
199                                     {
200                                         colorClass = "labColor";
201                                     }
202                                     else if (("red" in value) && ("green" in value) && ("blue" in value))
203                                     {
204                                         colorClass = "RGBColor";
205                                     }
206                                     typedValue = [ "<object>", [ colorClass, restoreDesc (value) ] ];
207                                     break;
208                                 default:
209                                     typedValue = null;
210                                     break;
211                             }
212                             if (typedValue)
213                             {
214                                 restoredDesc[key] = typedValue;
215                             }
216                         }
217                     }
218                     return restoredDesc;
219                 }

247                                     colorObject =
248                                     [
249                                         "<object>",
250                                         [
251                                             "bookColor",
252                                             {
253                                                 "bookID": [ "<integer>", color[1][0] ],
254                                                 "bookKey": [ "<data>", color[1][1] ]
255                                             }
256                                         ]
257                                     ];
258                                 }
259                                 break;
260                             case 4:
261                                 colorObject =
262                                 [
263                                     "<object>",
264                                     [
265                                         "bookColor",
266                                         {
267                                             "book": [ "<string>", color[1][0] ],
268                                             "name": [ "<string>", color[1][1] ],
269                                             "bookID": [ "<integer>", color[1][2] ],
270                                             "bookKey": [ "<data>", color[1][3] ]
271                                         }
272                                     ]
273                                 ];
274                                 break;
275                         }
276                         break;
277                     case jamEngine.uniIdStrToId ("CMYKColorClass"):
278                         colorObject =
279                         [
280                             "<object>",
281                             [
282                                 "CMYKColorClass",
283                                 {
284                                     "cyan": [ "<double>", color[1][0] ],

294                         [
295                             "<object>",
296                             [
297                                 "grayscale",
298                                 {
299                                     "gray": [ "<double>", (color[1].constructor === Array) ? color[1][0] : color[1] ]
300                                 }
301                             ]
302                         ];
303                         break;
304                     case jamEngine.uniIdStrToId ("HSBColorClass"):
305                         colorObject =
306                         [
307                             "<object>",
308                             [
309                                 "HSBColorClass",

354         /**
355          * @description Get a simplified color JSON object or JSON array from a color object in JSON AM Data Format.
356          * @param {Object|Array} colorObject Color object in <a href="http://www.tonton-pixel.com/blog/json-photoshop-scripting/json-am-data-format/">JSON AM Data Format</a>
357          * @param {Boolean} [explicit] Explicit flag: if true, return an explicit JSON object instead of a minimal JSON array
358          * @returns {Object|Array} Simplified color JSON object or JSON array
359          * (cf. <a href="http://www.tonton-pixel.com/blog/json-photoshop-scripting/json-simplified-formats/color-object-simplified-format/">Color Object Simplified Format</a>)
360          * @see jamHelpers.toColorObject
361          * @example
362          * jamEngine.meaningfulIds = true;
363          * var resultObj = jamEngine.jsonGet ([ { "application": { "<enumerated>": { "ordinal": "targetEnum" } } } ]);
364          * var fgColorArr = jamHelpers.<strong>fromColorObject</strong> (resultObj["foregroundColor"], true);
365          * alert (jamJSON.stringify (fgColorArr));
366          * // -> { "hue": 260, "saturation": 48, "brightness": 92 } // Lavender
367          * var bgColorArr = jamHelpers.<strong>fromColorObject</strong> (resultObj["backgroundColor"]);
368          * alert (jamJSON.stringify (bgColorArr));
369          * // -> [ "bookColor", [ "PANTONE® solid coated", "PANTONE 2395 C", 3002, " 2395C" ] ] // Vivid Magenta
370          */
371         jamHelpers.fromColorObject = function (colorObject, explicit)
372         {
373             var color;
374             if (explicit)
375             {
376                 color = jamEngine.simplifyObject (colorObject);
377             }
378             else
379             {
380                 var normalizedColorObject = jamEngine.normalizeJsonItem (colorObject, { meaningfulIds: true, parseFriendly: true });
381                 var colorClass = normalizedColorObject[1][0];
382                 var colorDesc = normalizedColorObject[1][1];
383                 switch (colorClass)
384                 {
385                     case "bookColor":
386                         var book = colorDesc["book"][1];
387                         var name = colorDesc["name"][1];
388                         if (("bookID" in colorDesc) && ("bookKey" in colorDesc))
389                         {
390                             var bookID = colorDesc["bookID"][1];
391                             var bookKey = colorDesc["bookKey"][1];
392                             color = [ colorClass, [ book, name, bookID, bookKey ] ];
393                         }
394                         else
395                         {
396                             color = [ colorClass, [ book, name ] ];
397                         }
398                         break;
399                     case "CMYKColorClass":
400                         var cyan = colorDesc["cyan"][1];
401                         var magenta =  colorDesc["magenta"][1];
402                         var yellowColor = colorDesc["yellowColor"][1];
403                         var black = colorDesc["black"][1];
404                         color = [ colorClass, [ cyan, magenta, yellowColor, black ] ];
405                         break;
406                     case "grayscale":
407                         var gray = colorDesc["gray"][1];
408                         color = [ colorClass, [ gray ] ];
409                         break;
410                     case "HSBColorClass":
411                         var hue = colorDesc["hue"][1][1];
412                         var saturation = colorDesc["saturation"][1];
413                         var brightness = colorDesc["brightness"][1];
414                         color = [ colorClass, [ hue, saturation, brightness ] ];
415                         break;
416                     case "labColor":
417                         var luminance = colorDesc["luminance"][1];
418                         var a = colorDesc["a"][1];
419                         var b = colorDesc["b"][1];
420                         color = [ colorClass, [ luminance, a, b ] ];
421                         break;
422                     case "RGBColor":
423                         var red =  colorDesc["red"][1];
424                         var green = colorDesc["green"][1];
425                         var blue = colorDesc["blue"][1];
426                         color = [ colorClass, [ red, green, blue ] ];
427                         break;
428                     default:
429                         throw new Error ("\n[jamHelpers.fromColorObject] Unrecognized color class: " + colorClass);
430                         break;
431                 }
432             }
433             return color;

494          *         "to": jamHelpers.toPointObject ([ [ 75.0, 50.0 ], "percentUnit" ]),
495          *         "opacity": { "<unitDouble>": { "percentUnit": 100 } },
496          *         "mode": { "<enumerated>": { "blendMode": "normal" } },
497          *         "type": { "<enumerated>": { "gradientType": "linear" } },

501          */

503         {
504             var gradientObject;
505             if (gradient.constructor === Object)
506             {
507                 var that = this;
508                 function restoreDesc (desc)
509                 {
510                     var restoredDesc = { };
511                     for (var key in desc)
512                     {
513                         if (desc.hasOwnProperty (key))
514                         {
515                             var value = desc[key];
516                             var typedValue;
517                             var restoredList;
518                             switch (key)
519                             {
520                                 case "showTransparency":
521                                 case "vectorColor":
522                                     typedValue = [ "<boolean>", value ];
523                                     break;
524                                 case "name":
525                                     typedValue = [ "<string>", localize (value) ];
526                                     break;
527                                 case "gradientForm":
528                                     typedValue = [ "<enumerated>", [ "gradientForm", value ] ];
529                                     break;
530                                 case "type":
531                                     typedValue = [ "<enumerated>", [ "colorStopType", value ] ];
532                                     break;
533                                 case "colorSpace":
534                                     typedValue = [ "<enumerated>", [ "colorSpace", value ] ];
535                                     break;
536                                 case "location":
537                                 case "midpoint":
538                                 case "randomSeed":
539                                 case "smoothness":
540                                     typedValue = [ "<integer>", value ];
541                                     break;
542                                 case "interpolation":
543                                     typedValue = [ "<double>", value ];
544                                     break;
545                                 case "opacity":
546                                     typedValue = [ "<unitDouble>", [ "percentUnit", value ] ];
547                                     break;
548                                 case "colors":
549                                     restoredList = [ ];
550                                     for (var i = 0; i < value.length; i++)
551                                     {
552                                         restoredList.push ([ "<object>", [ "colorStop", restoreDesc (value[i]) ] ]);
553                                     }
554                                     typedValue = [ "<list>", restoredList ];
555                                     break;
556                                 case "transparency":
557                                     restoredList = [ ];
558                                     for (var i = 0; i < value.length; i++)
559                                     {
560                                         restoredList.push ([ "<object>", [ "transparencyStop", restoreDesc (value[i]) ] ]);
561                                     }
562                                     typedValue = [ "<list>", restoredList ];
563                                     break;
564                                 case "minimum":
565                                 case "maximum":
566                                     restoredList = [ ];
567                                     for (var i = 0; i < value.length; i++)
568                                     {
569                                         restoredList.push ([ "<integer>", value[i] ]);
570                                     }
571                                     typedValue = [ "<list>", restoredList ];
572                                     break;
573                                 case "color":
574                                     typedValue = that.toColorObject (value);
575                                     break;
576                                 case "gradient":
577                                     typedValue = [ "<object>", [ "gradientClassEvent", restoreDesc (value) ] ];
578                                     break;
579                                 default:
580                                     typedValue = null;
581                                     break;
582                             }
583                             if (typedValue)
584                             {
585                                 restoredDesc[key] = typedValue;
586                             }
587                         }
588                     }
589                     return restoredDesc;
590                 }
591                 gradientObject = restoreDesc ({ "gradient": gradient })["gradient"];
592             }
593             else if (gradient.constructor === Array)
594             {
595                 var gradientObj = { };
596                 var gradientName = gradient[0];  // Name
597                 if (gradientName)   // Can be null (or empty string)
598                 {
599                     gradientObj["name"] = [ "<string>", gradientName ];
600                 }
601                 var gradientForm = gradient[1];  // Gradient Type
602                 gradientObj["gradientForm"] = [ "<enumerated>", [ "gradientForm", gradientForm ] ];
603                 switch (jamEngine.uniIdStrToId (gradientForm))
604                 {
605                     case jamEngine.uniIdStrToId ("customStops"):    // Solid
606                         gradientObj["interpolation"] = [ "<double>", gradient[2] ];  // Smoothness
607                         var colorStops = gradient[3];
608                         var colorsArr = [ ];
609                         for (var i = 0; i < colorStops.length; i++)
610                         {
611                             var colorStopObj = { };
612                             colorStopObj["location"] = [ "<integer>", colorStops[i][0] ];
613                             colorStopObj["midpoint"] = [ "<integer>", colorStops[i][1] ];
614                             var type = colorStops[i][2];
615                             colorStopObj["type"] = [ "<enumerated>", [ "colorStopType", type ] ];
616                             switch (jamEngine.uniIdStrToId (type))
617                             {
618                                 case jamEngine.uniIdStrToId ("userStop"):
619                                     colorStopObj["color"] = this.toColorObject (colorStops[i][3]);
620                                     break;
621                                 case jamEngine.uniIdStrToId ("backgroundColor"):
622                                 case jamEngine.uniIdStrToId ("foregroundColor"):
623                                     break;
624                                 default:
625                                     throw new Error ("\n[jamHelpers.toGradientObject] Unrecognized color stop type: " + type);
626                                     break;
627                             }
628                             colorsArr.push ([ "<object>", [ "colorStop", colorStopObj ] ]);
